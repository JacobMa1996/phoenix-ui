import {FormGroup,Row,Col,Input} from '../../src/index';

export default class input extends Component{

    constructor(props,context){

        super(props,context);

            name:''
        };

        this.setState(o);

    render(){

                        <Col>

                            <label for="">兴趣爱好</label>
